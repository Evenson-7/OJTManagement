// --- Cloudinary Configuration ---
const cloudName = "dixpsqyhx";
const uploadPreset = "Profile"; 

// --- Color Utility Classes ---
const ACCENT_TEXT = 'text-[#0094FF]'; // Deep Blue Accent
const PRIMARY_FOCUS_RING = 'focus:ring-[#0094FF]';
const LIGHT_BG = 'bg-[#BDE4F7]'; // Light Cyan Tint

// --- Utility Functions ---

const uploadImageToCloudinary = async (file) => {
  const formData = new FormData();
  formData.append("file", file);
  formData.append("upload_preset", uploadPreset);

  try {
    const response = await fetch(
      `https://api.cloudinary.com/v1_1/${cloudName}/image/upload`,
      {
        method: "POST",
        body: formData,
      }
    );
    const data = await response.json();
    if (data.secure_url) {
      return data.secure_url;
    } else {
      throw new Error(data.error?.message || "Cloudinary upload failed");
    }
  } catch (error) {
    console.error("Cloudinary upload error:", error);
    toast.error(`Upload failed: ${error.message}`);
    return null;
  }
};

const getCurrentYear = () => new Date().getFullYear();

// Utility function to count words
const countWords = (text) => {
  if (!text) return 0;
  return text.trim().split(/\s+/).filter(word => word.length > 0).length;
};

// --- Month/Year Options ---
const MONTH_OPTIONS = [
  { value: '01', name: 'January' }, { value: '02', name: 'February' },
  { value: '03', name: 'March' }, { value: '04', name: 'April' },
  { value: '05', name: 'May' }, { value: '06', name: 'June' },
  { value: '07', name: 'July' }, { value: '08', name: 'August' },
  { value: '09', name: 'September' }, { value: '10', name: 'October' },
  { value: '11', name: 'November' }, { value: '12', name: 'December' },
];

const getYearOptions = () => {
  const currentYear = getCurrentYear();
  const years = [];
  for (let i = 0; i < 3; i++) {
    years.push(currentYear - i);
  }
  return years.map(year => ({ value: String(year), name: String(year) }));
};

const formatDate = (timestamp) => {
    let date;
    if (timestamp?.toDate) {
        date = timestamp.toDate();
    } else if (timestamp) {
        date = new Date(timestamp);
    } else {
        return 'Invalid Date';
    }
    
    return date.toLocaleDateString('en-US', {
        year: 'numeric',
        month: 'short',
        day: 'numeric',
        hour: '2-digit',
        minute: '2-digit',
    });
};

const getStatusBadge = (status) => {
    if (status === 'approved') {
        return (
            <span className="flex items-center gap-1.5 px-3 py-1 rounded-full text-xs font-medium bg-green-50 text-green-700 border border-green-200">
                <Check size={14} />
                Approved
            </span>
        );
    }
    return (
        <span className="flex items-center gap-1.5 px-3 py-1 rounded-full text-xs font-medium bg-yellow-50 text-yellow-700 border border-yellow-200">
            <Clock size={14} />
            Pending
        </span>
    );
};

const getReportTypeColor = (type) =>
    type === 'weekly'

        }

        const newImages = [...form.images];

        setEditingReport(null);

            year: String(getCurrentYear())

                        Report Type

                    <div className="relative">
                        <select
                            id="report-type"
                            value={form.type}
                            onChange={(e) => setForm(prev => ({ ...prev, type: e.target.value }))}
                            className={`w-full appearance-none px-4 py-3 border border-gray-300 rounded-lg focus:ring-2 ${PRIMARY_FOCUS_RING} focus:border-transparent bg-white pr-10`}

                            disabled={!!editingReport}

                            <option value="weekly">Weekly Report</option>

                    </div>

                    <label htmlFor="title" className="block text-sm font-medium text-gray-700 mb-2">
                        Report Title
                    </label>
                    <input
                        type="text"
                        id="title"
                        value={form.title}
                        onChange={(e) => setForm(prev => ({ ...prev, title: e.target.value }))}

                    <span>View</span>
                </button>
                {isSupervisor && report.status !== 'approved' && (
                    <button
                        onClick={() => handleApprove(report.id)}
                        className="flex items-center gap-1.5 px-3 py-2 text-xs font-medium text-green-600 hover:text-green-700 hover:bg-green-50 rounded-lg transition-colors"
                    >
                        <Check size={16} />
                        <span>Approve</span>
                    </button>
                )}
            </div>
        </div>
        <div className={`bg-gray-50 rounded-lg p-4 border-l-4 border-[#0094FF]`}>
            <p className="text-gray-700 whitespace-pre-wrap leading-relaxed line-clamp-3">{report.content}</p>
            {report.images && report.images.length > 0 && (
                <div className="mt-4 pt-3 border-t border-gray-200 flex flex-wrap gap-3">
                    {report.images.map((imgUrl, index) => (
                        <a 
                          key={index} 
                          href={imgUrl} 
                          target="_blank" 
                          rel="noopener noreferrer"
                          className="w-20 h-20 bg-gray-200 rounded-lg overflow-hidden shadow-sm transition-transform hover:scale-105"
                        >
                            <img src={imgUrl} alt={`Report image ${index + 1}`} className="w-full h-full object-cover" />
                        </a>
                    ))}
                </div>
            )}
        </div>
    </div>
);

    return (
        <div className="fixed inset-0 z-50 flex items-center justify-center bg-black/50 bg-opacity-75 p-4">

                    <button

                        <X size={24} />
                    </button>
                </div>
                <div className="p-6 overflow-y-auto">
                    <div className="flex flex-wrap gap-4 text-sm text-gray-500 mb-4">
                        <div className="flex items-center gap-1.5">
                            <User size={14} />
                            <span className="font-medium text-gray-700">{report.userName}</span>
                        </div>
                        <div className="flex items-center gap-1.5">
                            <Calendar size={14} />
                            <span>{report.date}</span>
                        </div>
                        <div className="flex items-center gap-1.5">
                            <Clock size={14} />
                            <span>{formatDate(report.createdAt)}</span>
                        </div>
                        <div className="ml-auto">
                            {getStatusBadge(report.status)}
                        </div>
                    </div>
                    <div className="bg-gray-50 rounded-lg p-6 border border-gray-200">
                        <h4 className="text-sm font-semibold text-gray-700 mb-2 uppercase">Report Content</h4>
                        <p className="text-gray-800 whitespace-pre-wrap leading-relaxed">
                            {report.content}
                        </p>
                    </div>
                    {report.images && report.images.length > 0 && (
                        <div className="mt-6">
                            <h4 className="text-sm font-semibold text-gray-700 mb-3 uppercase">Supporting Images</h4>
                            <div className="flex flex-wrap gap-4">
                                {report.images.map((imgUrl, index) => (
                                    <a 
                                      key={index} 
                                      href={imgUrl} 
                                      target="_blank" 
                                      rel="noopener noreferrer"
                                      className="w-40 h-40 bg-gray-200 rounded-lg overflow-hidden shadow-sm transition-transform hover:scale-105"
                                    >
                                        <img src={imgUrl} alt={`Report image ${index + 1}`} className="w-full h-full object-cover" />
                                    </a>
                                ))}
                            </div>
                        </div>
                    )}
                </div>
                <div className="p-4 bg-gray-50 border-t border-gray-200 flex justify-end">
                    <button
                        onClick={onClose}
                        className="px-4 py-2 bg-gray-600 text-white rounded-lg hover:bg-gray-700"
                    >
                        Close
                    </button>
                </div>
            </div>
        </div>
    );
};


function ReportsTab({ user, reports: initialReports = [], onCreateReport, onDeleteReport }) {
    const [myReports, setMyReports] = useState(initialReports);
    const [myReportsLoading, setMyReportsLoading] = useState(true);

    const [form, setForm] = useState({
        title: '',
        content: '',
        type: 'weekly',
        startDate: new Date(), 
        endDate: new Date(),  
        month: String(new Date().getMonth() + 1).padStart(2, '0'),
        year: String(getCurrentYear()),
        images: [null, null, null], 
    });
    const [submitting, setSubmitting] = useState(false);
    const [activeView, setActiveView] = useState('create');
    const [editingReport, setEditingReport] = useState(null);
    const [viewingReport, setViewingReport] = useState(null);

    const [allReports, setAllReports] = useState([]);
    const [filteredReports, setFilteredReports] = useState([]);
    const [searchTerm, setSearchTerm] = useState('');
    const [selectedIntern, setSelectedIntern] = useState('');
    const [interns, setInterns] = useState([]);
    const [loading, setLoading] = useState(false);
    const [reportTypeFilter, setReportTypeFilter] = useState('all');

    const handleEdit = (report) => {
        setEditingReport(report);
        let startDate = new Date();
        let endDate = new Date();
        let month = String(new Date().getMonth() + 1).padStart(2, '0');
        let year = String(getCurrentYear());

        if (report.type === 'weekly' && report.date.includes(' to ')) {
            const dates = report.date.split(' to ');
            startDate = new Date(dates[0] + 'T00:00:00');
            endDate = new Date(dates[1] + 'T00:00:00');
        } else if (report.type === 'monthly' && report.date.includes('/')) {
            const parts = report.date.split('/');
            month = parts[0];
            year = parts[1];
        }

        setForm({
            title: report.title,
            content: report.content,
            type: report.type,
            startDate,
            endDate,
            month,
            year,
            images: [
                (report.images && report.images[0]) || null, 
                (report.images && report.images[1]) || null, 
                (report.images && report.images[2]) || null
            ],
        });
        setActiveView('create');
    };

    const handleView = (report) => {
        setViewingReport(report);
    };

    const handleApprove = async (reportId) => {
        try {
            const reportRef = doc(db, 'reports', reportId);
            await updateDoc(reportRef, {
                status: 'approved'
            });
            toast.success("Report approved!");
        } catch (error) {
            console.error("Error approving report:", error);
            toast.error("Failed to approve report.");
        }
    };

    useEffect(() => {
        if (user?.role === 'intern' && user.uid) {
            setMyReportsLoading(true);
            const q = query(
                collection(db, 'reports'), 
                where('userId', '==', user.uid), 
                orderBy('createdAt', 'desc')
            );
            
            const unsubscribe = onSnapshot(q, (querySnapshot) => {
                const fetchedReports = [];
                querySnapshot.forEach((doc) => {
                    fetchedReports.push({ id: doc.id, ...doc.data() });
                });
                setMyReports(fetchedReports);
                setMyReportsLoading(false);
            }, (error) => {
                console.error("Error fetching reports: ", error);
                toast.error("Failed to load reports.");
                setMyReportsLoading(false);
            });

            return () => unsubscribe(); 
        } else {
            setMyReportsLoading(false);
        }
    }, [user]);
    
    const loadAllReports = useCallback(() => {
        if (!user?.uid || user.role !== 'supervisor') return;
        setLoading(true);

        const internsQuery = query(
            collection(db, 'users'),
            where('role', '==', 'intern'),
            where('supervisorId', '==', user.uid)
        );

        getDocs(internsQuery).then(internsSnap => {
            const internIds = internsSnap.docs.map((doc) => doc.id);
            const internNames = internsSnap.docs.map(
                (doc) => getSafeName(doc.data()) 
            );
            setInterns(internNames.sort());

            if (internIds.length === 0) {
                setAllReports([]);
                setLoading(false);
                return null; 
            }
            
            let reportsQuery;
            if (internIds.length <= 30) {
                 reportsQuery = query(
                    collection(db, 'reports'), 
                    where('userId', 'in', internIds),
                    orderBy('createdAt', 'desc')
                );
            } else {
                reportsQuery = query(
                    collection(db, 'reports'),
                    orderBy('createdAt', 'desc')
                );
            }

            const unsubscribe = onSnapshot(reportsQuery, (querySnapshot) => {
                const data = [];
                querySnapshot.forEach((doc) => {
                    if (internIds.length > 30 && !internIds.includes(doc.data().userId)) {
                        return;
                    }
                    data.push({ id: doc.id, ...doc.data() });
                });
                setAllReports(data);
                setLoading(false);
            }, (error) => {
                console.error('Error loading supervisor reports:', error);
                setAllReports([]);
                setLoading(false);
            });

            return unsubscribe;
        }).catch(error => {
             console.error('Error fetching interns for reports:', error);
             setLoading(false);
        });

    }, [user]);

    useEffect(() => {
        if (user?.role === 'supervisor') {
            const unsubscribe = loadAllReports();
            return () => {
                if (unsubscribe) unsubscribe();
            };
        }
    }, [user, loadAllReports]);


    const handleSubmit = async (e) => {
        e.preventDefault();
        
        if (user.role === 'intern' && !user.supervisorId) {
            toast.error("You cannot submit a report until a supervisor is assigned to you.");
            return;
        }

        // --- UPDATED: Min words check ---
        const minWords = 100; // Changed from 200 to 100
        const wordCount = countWords(form.content);

        if (!form.title.trim() || !form.content.trim()) return;
        if (wordCount < minWords) {
            toast.error(`Report is too short. Please write at least ${minWords} words.`);
            return;
        }

        const uploadedImagesCount = form.images.filter(img => img).length;
        if (uploadedImagesCount !== 3) {
            toast.error("You must upload exactly 3 supporting images.");
            return;
        }
        
        let reportPeriod = '';
        if (form.type === 'weekly') {
            const startString = form.startDate.toISOString().slice(0, 10);
            const endString = form.endDate.toISOString().slice(0, 10);
            reportPeriod = `${startString} to ${endString}`;
        } else {
            reportPeriod = `${form.month}/${form.year}`;
        }

        setSubmitting(true);
        try {
            
            if (editingReport) {
                // UPDATE
                const updateData = {
                    title: form.title,
                    content: form.content,
                    type: form.type,
                    date: reportPeriod,
                    images: form.images.filter(img => img),
                    updatedAt: serverTimestamp(),
                    
                    userId: editingReport.userId,
                    userName: editingReport.userName,
                    status: editingReport.status 
                };
                
                const reportRef = doc(db, 'reports', editingReport.id);
                await updateDoc(reportRef, updateData); 
                toast.success("Report updated successfully!");

            } else {
                // CREATE
                const createData = {
                    title: form.title, 
                    content: form.content, 
                    type: form.type, 
                    date: reportPeriod, 
                    userId: user.uid, 
                    userName: user.firstName && user.lastName ? `${user.firstName} ${user.lastName}` : (user.name || user.fullName || user.email || 'Intern'),
                    images: form.images.filter(img => img),
                    status: 'pending', 
                    createdAt: serverTimestamp()
                };
                
                await addDoc(collection(db, 'reports'), createData);
                toast.success("Report submitted successfully!");
            }
            
            setForm(prev => ({
                ...prev, title: '', content: '', images: [null, null, null], 
                startDate: new Date(), endDate: new Date(),
                month: String(new Date().getMonth() + 1).padStart(2, '0'),
                year: String(getCurrentYear())
            }));
            setEditingReport(null);
            setActiveView('view'); 
            
        } catch (error) {
            console.error('Failed to submit report:', error);
            toast.error('Failed to submit report. Please try again.');
        } finally {
            setSubmitting(false);
        }
    };
    
    const handleDelete = async (reportId) => {
        if (!confirm('Are you sure you want to delete this report?')) return;
        try {
            await deleteDoc(doc(db, 'reports', reportId));
            toast.success("Report deleted successfully.");
        } catch (error) {
            console.error('Error deleting report:', error);
            toast.error('Error deleting report.');
        }
    };

    const filterReports = useCallback(() => {
        let filtered = [...allReports];

        if (searchTerm.trim()) {
            filtered = filtered.filter((report) =>
                report.userName?.toLowerCase().includes(searchTerm.toLowerCase())
            );
        }

        if (selectedIntern) {
            filtered = filtered.filter((report) => report.userName === selectedIntern);
        }

        if (reportTypeFilter !== 'all') {
            filtered = filtered.filter((report) => report.type === reportTypeFilter);
        }

        setFilteredReports(filtered);
    }, [allReports, searchTerm, selectedIntern, reportTypeFilter]);

    useEffect(() => {
        if(user.role === 'supervisor') {
            filterReports();
        }
    }, [filterReports, user.role]);

    const clearFilters = () => {
        setSearchTerm('');
        setSelectedIntern('');
        setReportTypeFilter('all');
    };
    
    const getSafeName = (intern) => {
      if (!intern) return 'Unknown Intern';
      if (intern.firstName && intern.lastName) {
        return `${intern.firstName} ${intern.lastName}`;
      }
      const name = intern.internName || intern.fullName || intern.name || intern.displayName;
      if (!name || name.includes('@')) {
        return 'Unknown Intern'; 
      }
      return name;
    };


    if (user?.role === 'intern') {
        return (
            <>
                <Toaster position="top-right" />
                <ViewReportModal report={viewingReport} onClose={() => setViewingReport(null)} />
                <div className="space-y-6">
                    <h1 className="text-3xl font-bold text-gray-900">My Reports</h1>
                    <p className="text-gray-600 mt-2">Submit your weekly and monthly reports</p>
                    
                    <nav className="flex space-x-2 bg-white rounded-lg p-1 shadow-sm border border-gray-200 max-w-fit">
                        <button
                            onClick={() => setActiveView('create')}
                            className={`flex items-center space-x-2 px-4 py-2 rounded-md font-medium transition-all ${
                                activeView === 'create'
                                    ? 'bg-[#0094FF] text-white shadow-md'
                                    : 'text-gray-600 hover:text-gray-900 hover:bg-gray-50'
                            }`}
                        >
                            {editingReport ? <Edit size={18} /> : <Plus size={18} />}
                            <span>{editingReport ? 'Edit Report' : 'Create Report'}</span>
                        </button>
                        <button
                            onClick={() => {
                                setActiveView('view');
                                setEditingReport(null); 
                            }}
                            className={`flex items-center space-x-2 px-4 py-2 rounded-md font-medium transition-all ${
                                activeView === 'view'
                                    ? 'bg-[#0094FF] text-white shadow-md'
                                    : 'text-gray-600 hover:text-gray-900 hover:bg-gray-50'
                            }`}
                        >
                            <FileText size={18} />
                            <span>My Submissions</span>
                            <span className={`ml-1 px-2 py-0.5 bg-white ${ACCENT_TEXT} rounded-full text-xs font-semibold shadow-inner`}>
                                {myReports.length}
                            </span>
                        </button>
                    </nav>

                    {activeView === 'create' && (
                        <CreateReportForm 
                            user={user} 
                            form={form} 
                            setForm={setForm} 
                            handleSubmit={handleSubmit} 
                            submitting={submitting} 
                            reports={myReports}
                            editingReport={editingReport}
                            setEditingReport={setEditingReport}
                        />
                    )}

                    {activeView === 'view' && (
                        myReportsLoading ? (
                             <div className="text-center py-16 bg-white rounded-xl border border-gray-200 shadow-sm">
                                <Loader2 size={32} className="text-gray-400 mx-auto mb-4 animate-spin" />
                                <p className="text-gray-500">Loading reports...</p>
                            </div>
                        ) : (
                            <MySubmissionsList 
                                reports={myReports} 
                                handleDelete={handleDelete}
                                handleEdit={handleEdit}
                                handleView={handleView}
                            />
                        )
                    )}
                </div>
            </>
        );
    }

    return (
        <>
            <Toaster position="top-right" />
            <ViewReportModal report={viewingReport} onClose={() => setViewingReport(null)} />
            <SupervisorReportManagement
                loading={loading}
                filteredReports={filteredReports}
                searchTerm={searchTerm}
                interns={interns}
                selectedIntern={selectedIntern}
                reportTypeFilter={reportTypeFilter}
                setReportTypeFilter={setReportTypeFilter}
                setSearchTerm={setSearchTerm}
                setSelectedIntern={setSelectedIntern}
                clearFilters={clearFilters}
                handleApprove={handleApprove}
                handleView={handleView}
            />
        </>
    );
}

export default ReportsTab;
