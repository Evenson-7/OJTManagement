// --- Utility Functions ---
const getSafeName = (intern) => {
  if (!intern) return 'Unknown';
  return intern.internName || intern.fullName || intern.name || intern.displayName || intern.email || 'Unknown';
};

const getFirstChar = (name) => {
  const safeName = getSafeName({ internName: name });
  return safeName.charAt(0).toUpperCase();
};

// --- Helper Components ---

// Helper function to get trend icon
function getTrendIcon(trend) {
  if (trend === 'Improving') return <HiArrowTrendingUp className="w-5 h-5 text-green-500" />;
  if (trend === 'Declining') return <HiArrowTrendingDown className="w-5 h-5 text-red-500" />;
  return <HiArrowRight className="w-5 h-5 text-gray-400" />;
}

// Helper component for the stats cards
function StatCard({ title, value, icon, change, changeType }) {
  const ACCENT_COLOR = 'text-[#0094FF]'; 
  const LIGHT_BG = 'bg-[#BDE4F7]'; 

  return (
    <div className="bg-white rounded-lg border border-gray-200 p-5 shadow-sm">
      <div className="flex items-center justify-between">
        <div>
          <p className="text-sm font-medium text-gray-500">{title}</p>
          <p className="text-3xl font-bold text-gray-900">{value}</p>
        </div>
        <div className={`${LIGHT_BG} ${ACCENT_COLOR} p-3 rounded-full`}> 
          {icon}
        </div>
      </div>
      {change && (
        <div className={`mt-3 flex items-center text-sm ${
          changeType === 'increase' ? 'text-green-600' :
          changeType === 'decrease' ? 'text-red-600' : 'text-gray-500'
        }`}>
          {changeType === 'increase' ? <HiArrowTrendingUp className="w-5 h-5 mr-1" /> :
           changeType === 'decrease' ? <HiArrowTrendingDown className="w-5 h-5 mr-1" /> :
           <HiArrowRight className="w-5 h-5 mr-1" />}
          <span>{change}</span>
        </div>
      )}
    </div>
  );
}

// --- Main TabSection Component ---
const TabSection = ({ currentTab, data, user, handlers }) => {
  const {
    evaluations = [], myInterns = [], internRankings = [], myBadges = [], 
    performanceData = {}, evaluationForm = {}, loading = false, evaluationSubmitting = false,
    currentFormPage = 1, editingEvaluation = null, isEditMode = false, ratingScale = [],
    evaluationSections = {}, essayQuestions = [], badgeDefinitions = [],
    calculateSectionScore = () => 0, calculateOverallScore = () => 0,
    getPerformanceRating = () => ({ label: 'Not Rated', color: 'text-gray-600', bgColor: 'bg-gray-100' }),
    getSectionIcon = () => 'ðŸ“Š'
  } = data || {};

  const {
    handleEditEvaluation = () => {}, handleViewEvaluation = () => {}, handleDeleteEvaluation = () => {},
    resetEvaluationForm = () => {}, setActiveView = () => {}, setCurrentFormPage = () => {},
    handleRatingChange = () => {}, handleEssayChange = () => {}, handleInternSelection = () => {},
    handleSaveEvaluation = () => {}, handleFormChange = () => {},
    handleViewCertificate = () => {} 
  } = handlers || {};

  switch (currentTab) {
    case 'dashboard':
      return user?.role === 'supervisor' ? (
        <SupervisorDashboard 
          evaluations={evaluations} myInterns={myInterns} resetEvaluationForm={resetEvaluationForm} 
          setActiveView={setActiveView} handleViewEvaluation={handleViewEvaluation} 
          handleEditEvaluation={handleEditEvaluation} handleDeleteEvaluation={handleDeleteEvaluation}
        />
      ) : (
        <InternDashboard 
          evaluations={evaluations} 
          myBadges={myBadges} 
          setActiveView={setActiveView} 
          handleViewEvaluation={handleViewEvaluation}
          handleViewCertificate={handleViewCertificate}
        />
      );

    case 'rankings':
      return <RankingsSection internRankings={internRankings} user={user} />;

    case 'badges':
      return <BadgesSection myBadges={myBadges} badgeDefinitions={badgeDefinitions} />;

    case 'performance':
      return <PerformanceSection 
        evaluations={evaluations} performanceData={performanceData} user={user} 
        handleViewEvaluation={handleViewEvaluation}
        getSectionIcon={getSectionIcon} 
      />;

    case 'form':
      return <TwoPageEvaluationForm 
        evaluationForm={evaluationForm} myInterns={myInterns} loading={loading} 
        evaluationSubmitting={evaluationSubmitting} currentFormPage={currentFormPage} 
        editingEvaluation={editingEvaluation} isEditMode={isEditMode} resetEvaluationForm={resetEvaluationForm} 
        setActiveView={setActiveView} setCurrentFormPage={setCurrentFormPage} 
        handleRatingChange={handleRatingChange} handleEssayChange={handleEssayChange} 
        handleInternSelection={handleInternSelection} handleSaveEvaluation={handleSaveEvaluation} 
        calculateSectionScore={calculateSectionScore} calculateOverallScore={calculateOverallScore} 
        ratingScale={ratingScale} evaluationSections={evaluationSections} essayQuestions={essayQuestions} 
        user={user} data={data} 
        handleFormChange={handleFormChange} 
      />;

    default:
      return <div>Unknown section</div>;
  }
};

              ) : (
                <div className="px-4 py-3 bg-white border-2 border-gray-300 rounded-lg">
                  <strong className="text-gray-900">{evaluationForm.internName}</strong>
                </div>
              )}
              
              {!loading && myInterns.length === 0 && (
                <div className="mt-3 p-3 bg-amber-50 border border-amber-200 rounded-lg">
                  <p className="text-sm text-amber-800 flex items-center">
                    <HiOutlineExclamationTriangle className="w-4 h-4 mr-2" />
                    No interns assigned. Please use the Manage tab to assign interns first.
                  </p>
                </div>
              )}
            </div>
            
            {/* Period Covered (Month Range and Type Dropdown) */}
            <div className="md:col-span-2">
              <label className="block text-sm font-semibold text-gray-700 mb-3">
                Evaluation Month Range: <span className="text-red-500">*</span>
              </label>
              <div className="flex space-x-2 mb-3">
                {/* Start Month Dropdown */}
                <select
                  value={evaluationForm.periodStartMonth || ''}
                  onChange={(e) => handleFormChange('periodStartMonth', e.target.value)}
                  className={`w-1/2 px-4 py-3 border-2 border-gray-300 rounded-lg focus:ring-2 focus:ring-[#0094FF] focus:border-[#0094FF] transition-colors`}
                  required
                  disabled={evaluationSubmitting || (!isEditMode && editingEvaluation)}
                >
                  {monthOptions.map(option => (
                    <option key={`start-${option.value}`} value={option.value} disabled={!option.value && true}>
                      {option.value ? `${option.label} (Start)` : option.label}
                    </option>
                  ))}
                </select>
                {/* End Month Dropdown */}
                <select
                  value={evaluationForm.periodEndMonth || ''}
                  onChange={(e) => handleFormChange('periodEndMonth', e.target.value)}
                  className={`w-1/2 px-4 py-3 border-2 border-gray-300 rounded-lg focus:ring-2 focus:ring-[#0094FF] focus:border-[#0094FF] transition-colors`}
                  required
                  disabled={evaluationSubmitting || (!isEditMode && editingEvaluation)}
                >
                  {monthOptions.map(option => (
                    <option key={`end-${option.value}`} value={option.value} disabled={!option.value && true}>
                      {option.value ? `${option.label} (End)` : option.label}
                    </option>
                  ))}
                </select>
              </div>

              <label className="block text-sm font-semibold text-gray-700 mb-3">
                Period Covered by the Evaluation: <span className="text-red-500">*</span>
              </label>
              {/* Period Covered Dropdown */}
              <select
                value={evaluationForm.periodCovered || ''}
                onChange={(e) => handleFormChange('periodCovered', e.target.value)}
                className={`w-full px-4 py-3 border-2 border-gray-300 rounded-lg focus:ring-2 focus:ring-[#0094FF] focus:border-[#0094FF] transition-colors`}
                required
                disabled={evaluationSubmitting || (!isEditMode && editingEvaluation)}
              >
                {periodOptions.map(option => (
                  <option key={option.value} value={option.value} disabled={!option.value && true}>
                    {option.label}
                  </option>
                ))}
              </select>
            </div>
          </div>
        </div>
      </div>

      {(!evaluationForm.internId && !editingEvaluation) ? (
        <div className="text-center py-16 px-8">
          <div className="text-gray-400 mb-6 flex justify-center">
            <HiOutlineUser className="w-16 h-16" />
          </div>
          <h3 className="text-xl font-medium text-gray-900 mb-3">Select an Intern</h3>
          <p className="text-gray-500 mb-6 max-w-md mx-auto">
            Please select an intern from the dropdown above to start the evaluation process
          </p>
        </div>
      ) : (
        <div className="p-8">
          {editingEvaluation && (
            <div className={`${SUCCESS_BG} border ${ACCENT_BORDER} rounded-xl p-6 mb-8`}>
              <div className="flex items-center justify-center space-x-2">
                <HiOutlineUser className={`w-5 h-5 ${ACCENT_COLOR}`} />
                <p className={`${SUCCESS_TEXT} font-medium text-sm md:text-base`}>
                  <strong>Intern:</strong> {evaluationForm.internName} | 
                  <strong> Period:</strong> {evaluationForm.periodCovered} ({evaluationForm.periodStartMonth} - {evaluationForm.periodEndMonth}) |
                  <strong> Date:</strong> {evaluationForm.evaluationDate} |
                  <strong> Status:</strong> <span className="capitalize">{evaluationForm.status}</span>
                </p>
              </div>
            </div>
          )}

          {/* PAGE 1: Rating Sections */}
          {currentPage === 1 && (
            <div className="space-y-8">
              <div className={`${SUCCESS_BG} border ${ACCENT_BORDER} rounded-xl p-8`}>
                <h2 className={`text-2xl font-bold ${SUCCESS_TEXT} mb-6`}>
                  PART I. DIRECTIONS: Please encircle the letter that best evaluates the intern for the period specified above.
                </h2>
                <div className={`${SUCCESS_TEXT} mb-6 font-medium`}>Refer to the matrix below for interpretation:</div>
                
                {/* Rating Scale Legend */}
                <div className="grid grid-cols-1 md:grid-cols-5 gap-4 mb-8">
                  {ratingScale.map(rating => (
                    <div key={rating?.value} className="text-center p-4 bg-white rounded-xl border-2 border-gray-200 shadow-sm hover:shadow-md transition-shadow">
                      <div className="text-2xl font-bold text-gray-900 mb-1">{rating?.value}</div>
                      <div className="text-sm font-semibold text-gray-700 mb-1">{rating?.label}</div>
                      <div className="text-xs text-gray-500">({rating?.range})</div>
                    </div>
                  ))}
                </div>
              </div>

              {/* Evaluation Sections */}
              {Object.entries(evaluationSections).map(([sectionKey, section]) => (
                <div key={sectionKey} className="bg-gray-50 rounded-xl border border-gray-200 p-8 shadow-sm">
                  <h3 className="text-xl font-bold text-gray-900 mb-8 pb-4 border-b border-gray-300">{section?.title}</h3>
                  
                  <div className="space-y-6">
                    {(section?.items || []).map((item, index) => (
                      <div key={item?.id} className="bg-white p-6 rounded-xl border border-gray-200 shadow-sm hover:shadow-md transition-shadow">
                        <div className="flex items-center justify-between">
                          <div className="flex-1 pr-6">
                            <label className="text-base font-medium text-gray-900 cursor-pointer">
                              <span className={ACCENT_COLOR}>{String.fromCharCode(97 + index)}.</span> {item?.text}
                            </label>
                          </div>
                          
                          {/* Rating Buttons */}
                          <div className="flex space-x-4">
                            {ratingScale.map(rating => (
                              <label key={rating?.value} className="cursor-pointer group">
                                <input
                                  type="radio"
                                  name={`${sectionKey}_${item?.id}`}
                                  value={rating?.value}
                                  checked={(evaluationForm[sectionKey] || {})[item?.id] === rating?.value}
                                  onChange={() => handleRatingChange && handleRatingChange(sectionKey, item?.id, rating?.value)}
                                  disabled={evaluationSubmitting || (!isEditMode && editingEvaluation)}
                                  className="sr-only"
                                />
                                <div className={`w-12 h-12 rounded-full border-3 font-bold flex items-center justify-center text-base transition-all transform group-hover:scale-105 ${
                                  (evaluationForm[sectionKey] || {})[item?.id] === rating?.value
                                    ? `border-[#0094FF] ${SUCCESS_BG} ${ACCENT_COLOR} scale-110 shadow-lg ring-2 ring-[#BDE4F7]`
                                    : 'border-gray-300 hover:border-gray-400 hover:bg-gray-50 text-gray-600 group-hover:shadow-md'
                                } ${evaluationSubmitting || (!isEditMode && editingEvaluation) ? 'opacity-50 cursor-not-allowed' : ''}`}>
                                  {rating?.value}
                                </div>
                              </label>
                            ))}
                          </div>
                        </div>
                      </div>
                    ))}
                  </div>
                  
                  {/* --- UPDATED SECTION SCORE DISPLAY (1-5 Scale) --- */}
                  <div className={`mt-8 p-6 bg-white rounded-xl border-l-4 border-[#0094FF] shadow-sm`}>
                    <div className="flex justify-between items-center mb-3">
                      <span className="text-lg font-semibold text-gray-700">Section Average:</span>
                      <span className={`text-2xl font-bold ${ACCENT_COLOR}`}>
                        {/* Display raw average (e.g., 4.25) / 5.00 */}
                        {Number(safeCalculateSectionScore(evaluationForm[sectionKey])).toFixed(2)} / 5.00
                      </span>
                    </div>
                    <div className="w-full bg-gray-200 rounded-full h-4 overflow-hidden">
                      {/* Calculate Width based on scale of 5. e.g. Score 4 = 80% */}
                      <div 
                        className={`bg-gradient-to-r from-[#0094FF] to-[#42A5FF] h-4 rounded-full transition-all duration-700 ease-out`} 
                        style={{ width: `${(safeCalculateSectionScore(evaluationForm[sectionKey]) / 5) * 100}%` }}
                      ></div>
                    </div>
                    <p className="text-xs text-gray-500 mt-2 text-right">
                       * Calculated as average of all items in this section.
                    </p>
                  </div>
                </div>
              ))}
            </div>
          )}

          {/* PAGE 2: Essay Questions and Summary */}
          {currentPage === 2 && (
            <div className="space-y-8">
              
              {/* --- ADDED CRITERIA REFERENCE PANEL HERE --- */}
              {CriteriaReferencePanel && (
                 <div className="mb-6">
                    <CriteriaReferencePanel />
                 </div>
              )}

              {/* PART II: Essay Questions */}
              <div className="bg-gray-50 rounded-xl border border-gray-200 p-8 shadow-sm">
                <h2 className="text-2xl font-bold text-gray-900 mb-6 pb-4 border-b border-gray-300">
                  PART II. DIRECTIONS: Please answer the questions below.
                </h2>

                <div className="space-y-8">
                  {essayQuestions.map((question) => (
                    <div key={question?.id} className="bg-white rounded-xl border border-gray-200 p-8 shadow-sm">
                      <label className="block text-lg font-semibold text-gray-900 mb-4">
                        {question?.question}
                      </label>
                      <textarea
                        value={(evaluationForm.essayQuestions || {})[question?.id] || ""}
                        onChange={(e) => handleEssayChange && handleEssayChange(question?.id, e.target.value)}
                        placeholder={question?.placeholder}
                        rows={8}
                        disabled={evaluationSubmitting || (!isEditMode && editingEvaluation)}
                        className={`w-full px-4 py-4 border-2 border-gray-300 rounded-lg focus:ring-2 focus:ring-[#0094FF] focus:border-[#0094FF] resize-vertical transition-colors text-gray-800 leading-relaxed disabled:opacity-50 disabled:cursor-not-allowed`}
                      />
                    </div>
                  ))}
                </div>
              </div>

              {/* Signature Section */}
              <div className="bg-gray-50 rounded-xl border border-gray-200 p-8 shadow-sm">
                <h3 className="text-xl font-bold text-gray-900 mb-8 pb-4 border-b border-gray-300">
                  NAME & SIGNATURE OF SITE SUPERVISOR
                </h3>
                <div className="grid grid-cols-1 md:grid-cols-2 gap-8">
                  <div>
                    <label className="block text-sm font-semibold text-gray-700 mb-3">
                      Name of Site Supervisor:
                    </label>
                    <input
                      type="text"
                      value={evaluationForm.supervisorName || user?.name || user?.displayName || ""}
                      onChange={(e) => handleFormChange('supervisorName', e.target.value)}
                      className={`w-full px-4 py-4 bg-white border-b-4 border-gray-400 focus:border-[#0094FF] outline-none text-gray-900 font-semibold text-lg transition-colors`}
                      placeholder="Enter supervisor name"
                      disabled={evaluationSubmitting || (!isEditMode && editingEvaluation)}
                    />
                  </div>
                  <div>
                    <label className="block text-sm font-semibold text-gray-700 mb-3">
                      Date:
                    </label>
                    <input
                      type="date"
                      value={evaluationForm.evaluationDate || ''}
                      onChange={(e) => handleFormChange('evaluationDate', e.target.value)}
                      className={`w-full px-4 py-4 bg-white border-b-4 border-gray-400 focus:border-[#0094FF] outline-none text-gray-900 font-semibold text-lg transition-colors`}
                      disabled={evaluationSubmitting || (!isEditMode && editingEvaluation)}
                    />
                  </div>
                </div>
              </div>

              {/* --- UPDATED Overall Score Summary for 1-5 Scale --- */}
              <div className={`bg-gradient-to-r from-[#BDE4F7] to-[#42A5FF] border ${ACCENT_BORDER} rounded-xl p-8 shadow-lg`}>
                <h3 className={`text-xl font-bold ${SUCCESS_TEXT} mb-8 text-center`}>EVALUATION SUMMARY</h3>
                <div className="grid grid-cols-1 md:grid-cols-4 gap-6 text-center">
                  <div className="bg-white rounded-xl p-8 shadow-md border border-gray-200">
                    <div className={`flex justify-center ${ACCENT_COLOR} mb-3`}>
                        <HiOutlineChartPie className="w-10 h-10" />
                    </div>
                    {/* Display Raw Score / 5 */}
                    <div className={`text-4xl font-bold ${ACCENT_COLOR} mb-3`}>
                        {safeCalculateOverallScore().toFixed(2)} <span className="text-lg text-gray-400">/ 5</span>
                    </div>
                    <div className="text-base text-gray-600 mb-2 font-medium">Overall Average</div>
                    
                    {/* Updated Text Labels for 1-5 Scale */}
                    <div className="text-sm text-gray-500 font-bold">
                      {(() => {
                        const score = safeCalculateOverallScore();
                        if (score >= 4.5) return 'Outstanding (O)';
                        if (score >= 4.0) return 'Excellent (E)';
                        if (score >= 3.0) return 'Above Standard (A)';
                        if (score >= 2.0) return 'Standard (S)';
                        if (score >= 1.5) return 'Needs Improvement (N)';
                        return 'Poor (P)';
                      })()}
                    </div>
                  </div>
                  
                  <div className="bg-white rounded-xl p-8 shadow-md border border-gray-200">
                    <div className={`flex justify-center ${SUCCESS_TEXT} mb-3`}>
                        <HiOutlineCheck className="w-10 h-10" />
                    </div>
                    <div className={`text-5xl font-bold ${SUCCESS_TEXT} mb-3`}>
                      {Object.values(evaluationForm.behaviorAtWork || {}).filter(r => r).length + 
                       Object.values(evaluationForm.academicPerformance || {}).filter(r => r).length +
                       Object.values(evaluationForm.socialBehavior || {}).filter(r => r).length +
                       Object.values(evaluationForm.personality || {}).filter(r => r).length +
                       Object.values(evaluationForm.costConsciousness || {}).filter(r => r).length}
                    </div>
                    <div className="text-base text-gray-600 mb-2 font-medium">Items Rated</div>
                    <div className="text-sm text-gray-500">out of 33 total</div>
                  </div>
                  
                  <div className="bg-white rounded-xl p-8 shadow-md border border-gray-200">
                    <div className={`flex justify-center ${ACCENT_COLOR} mb-3`}>
                        <HiOutlinePencil className="w-10 h-10" />
                    </div>
                    <div className={`text-5xl font-bold ${ACCENT_COLOR} mb-3`}>
                      {Object.values(evaluationForm.essayQuestions || {}).filter(q => q?.trim()).length}
                    </div>
                    <div className="text-base text-gray-600 mb-2 font-medium">Questions Answered</div>
                    <div className="text-sm text-gray-500">out of 4 total</div>
                  </div>
                  
                  <div className="bg-white rounded-xl p-8 shadow-md border border-gray-200">
                    <div className={`flex justify-center ${DRAFT_TEXT} mb-3`}>
                        <HiOutlineCheckCircle className="w-10 h-10" />
                    </div>
                    <div className={`text-5xl font-bold ${DRAFT_TEXT} mb-3`}>
                      {Math.round(((Object.values(evaluationForm.behaviorAtWork || {}).filter(r => r).length + 
                                   Object.values(evaluationForm.academicPerformance || {}).filter(r => r).length +
                                   Object.values(evaluationForm.socialBehavior || {}).filter(r => r).length +
                                   Object.values(evaluationForm.personality || {}).filter(r => r).length +
                                   Object.values(evaluationForm.costConsciousness || {}).filter(r => r).length +
                                   Object.values(evaluationForm.essayQuestions || {}).filter(q => q?.trim()).length) / 37) * 100)}%
                    </div>
                    <div className="text-base text-gray-600 mb-2 font-medium">Completion</div>
                    <div className="text-sm text-gray-500">overall progress</div>
                  </div>
                </div>
              </div>
            </div>
          )}

          {/* Navigation Footer */}
          <div className="flex justify-between items-center pt-8 border-t border-gray-200">
            <div className="flex space-x-4">
              <button
                onClick={goToPreviousPage}
                disabled={currentPage === 1 || evaluationSubmitting}
                className={`px-8 py-4 border-2 border-gray-300 text-gray-700 rounded-xl font-medium transition-all disabled:opacity-50 disabled:cursor-not-allowed ${
                  currentPage === 1 ? 'invisible' : 'hover:bg-gray-50 hover:border-gray-400'
                }`}
              >
                â† Back
              </button>
            </div>

            <div className="text-center">
              <div className="text-sm text-gray-500 mb-2">Step {currentPage} of 2</div>
              <div className="w-64 bg-gray-200 rounded-full h-2">
                <div 
                  className={`${ACCENT_COLOR} h-2 rounded-full transition-all duration-300`} 
                  style={{ width: `${(currentPage / 2) * 100}%` }}
                ></div>
              </div>
            </div>

            <div className="flex space-x-4">
              {currentPage === 1 && (
                <button
                  onClick={goToNextPage}
                  disabled={evaluationSubmitting}
                  className={`${PRIMARY_ACTION_BG} text-white px-8 py-4 rounded-xl ${PRIMARY_ACTION_HOVER_BG} transition-all font-medium disabled:opacity-50 disabled:cursor-not-allowed`}
                >
                  Next â†’
                </button>
              )}

              {currentPage === 2 && (isEditMode || !editingEvaluation) && (
                <>
                  <button 
                    onClick={() => handleSaveEvaluation && handleSaveEvaluation("draft")}
                    disabled={evaluationSubmitting}
                    className="px-8 py-4 border-2 border-gray-300 text-gray-700 rounded-xl hover:bg-gray-50 hover:border-gray-400 transition-all font-medium disabled:opacity-50 disabled:cursor-not-allowed"
                  >
                    <HiOutlinePencilSquare className="w-5 h-5 mr-2 inline" />
                    {evaluationSubmitting ? "Saving..." : (isEditMode ? "Update Draft" : "Save as Draft")}
                  </button>
                  <button 
                    onClick={() => handleSaveEvaluation && handleSaveEvaluation("submitted")}
                    disabled={evaluationSubmitting}
                    className={`px-8 py-4 bg-gradient-to-r from-[#0094FF] to-[#002B66] text-white rounded-xl hover:from-[#002B66] hover:to-[#002B66] transition-all font-medium disabled:opacity-50 disabled:cursor-not-allowed shadow-lg hover:shadow-xl transform hover:scale-105`}
                  >
                    <HiOutlineCheck className="w-5 h-5 mr-2 inline" />
                    {evaluationSubmitting ? "Submitting..." : "Submit Evaluation"}
                  </button>
                </>
              )}
            </div>
          </div>
        </div>
      )}
    </div>
  );
};

export default TabSection;
